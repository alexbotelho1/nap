/**
* Contém dados de uma seção
* @constructor
*
* @author mpollack
* @version 1.0
* @param {Course} p referência ao objeto-pai
* @param {string} id ID da seção
* @param {string} ti título da seção
* @param {string} ty tipo de seção
* @param {int} nu número da seção
*
*/
function Section(p,id,ti,ty,nu) {
/**
* Nome fornecido para identificar o objeto na depuração 
* @type cadeia de caracteres
*/
	this.objname = "section";
/**
* Referência ao objeto-pai
* @type Capítulo
*/
	this.par = p;
/**
* ID da seção
* @type cadeia de caracteres
*/
	this.oid = id;
/**
* Título da seção
* @type cadeia de caracteres
*/
	this.title = ti;
/**
* Tipo de seção
* @type cadeia de caracteres
*/
	this.type = ty;
/**
* Número da seção
* @type cadeia de caracteres
*/
	this.num = nu;
/**
* Objetos de tópico em um arranjo seqüencial
* @type arranjo
*/
	this.topics = new Array();
/**
* Objetos de tópico em um arranjo associativo
* @type arranjo
*/
	this.topicsID = new Array();
}

/**
* Obtém o tópico de uma ID fornecida
* @param {string} i ID de tópico
* @return A {@link Topic} instância
* @type Tópico
*/
Section.prototype.getTopicByID = function(i) {
	var r = this.topicsID[i];
	return r;
}

/**
* Verifica a seção para ver se há mais alguma página na árvore
* contêm o meio de um determinado tipo
* @param {string} f tipo de filtro de meio
* @return booleano
* @type Booleano
*/
Section.prototype.containsMediaByFilter = function(f) {
	var check = false;
	for (var i=0; i<this.topics.length; i++) {
		var ct = this.topics[i];
		for (var j=0; jif (ct.pages[j].media.filtertype == f) {
				check = true;
				j = ct.pages.length;
				i = this.topics.length;
			}
		}
	}
	return check;
}

/**
* Envia informações de depuração de todas as propriedades para a janela de depuração
*/
Section.prototype.dump = function() {
	var d = "Section Dump\n";
	d += "\tthis.objname: "+this.objname+"\n";
	d += "\tthis.oid: "+this.oid+"\n";
	d += "\tthis.par: "+this.par+"\n";
	d += "\tthis.title: "+this.title+"\n";
	d += "\tthis.num: "+this.num+"\n";
	d += "\tthis.type: "+this.type+"\n";
	d += "\tthis.topics: "+this.topics+"\n";
	dump(d);
}
