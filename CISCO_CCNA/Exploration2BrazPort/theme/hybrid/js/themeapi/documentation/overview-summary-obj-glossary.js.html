/**
* Mantém todos os termos de glossário, conjuntos e controles de acesso
* @constructor
*
* @author mpollack
* @version 1.0
& @param {Main} p referência ao objeto-pai
*
*/
function Glossary(p) {
/**
* Nome fornecido para identificar o objeto na depuração 
* @type cadeia de caracteres
*/
	this.objname = "glossary";
/**
* Referência ao objeto-pai
* @type Principal
*/
	this.par = p;
/**
* Conjuntos de glossários em arranjo seqüencial
* @type arranjo
*/
	this.sets = new Array();
/**
* Conjuntos de glossários em arranjo associativo
* @type arranjo
*/
	this.setsID = new Array();
/**
* Termos de glossários em arranjo seqüencial
* @type arranjo
*/
	this.terms = new Array();
/**
* Termos de glossário em arranjo associativo
* @type arranjo
*/
	this.termsID = new Array();
}

/**
* Adiciona um objeto de termo aos arranjos de termo caso haja o conjunto
* adiciona esse termo ao arranjo do conjunto
* @param {string} t termo
* @param {string} d definição
* @param {string} i ID do termo
* @param {string} s ID do conjunto
* @param {string} sa consulte também termos
*/
Glossary.prototype.addTerm = function(t, d, i, s, sa) {
	var term = new GlossaryTerm(this, t, d, i, s, sa);
	this.terms.push(term);
	this.termsID[i] = term;
	if (this.setsID[s]) {
		this.setsID[s].addTerm(term);
	}
}
/**
* Adiciona um objeto de definição aos arranjos de definição
* @param {string} sk chave de definição
* @param {string} sl rótulo de definição
*/
Glossary.prototype.addSet = function(sk, sl) {
	var set = new GlossarySet(this, sk, sl);
	this.sets.push(set);
	this.setsID[sk] = set;
}
/**
* Retorna um objeto de definição com base na ID de definição ou em uma cadeia de caracteres de erro
* se a ID for inválida
* @param {string} i ID de definição
* @returns {GlossarySet} GlossarySet referência a objeto
*/
Glossary.prototype.getSetByID = function(i) {
	if (this.setsID[i]) {
		return this.setsID[i];
	} else {
		return "ERRO: Nenhum conjunto localizado com a ID: "+i;
	}
}
/**
* Retorna um objeto de termo com base na ID de definição ou em uma cadeia de caracteres de erro
* se a ID for inválida
* @param {string} i ID do termo
* @returns {GlossaryTerm} GlossaryTerm referência a objeto
*/
Glossary.prototype.getTermByID = function(i) {
	if (this.termsID[i]) {
		return this.termsID[i];
	} else {
		return "ERRO: Nenhum termo localizado com a ID: "+i;
	}
}
/**
* Envia informações de depuração sobre glossário, conjuntos e termos
* para a janela de depuração se ativada
*/
Glossary.prototype.dump = function() {
	var d = "Glossary Dump\n";
	d += "\tthis.objname: "+this.objname+"\n";
	d += "\tthis.par: "+this.par+"\n";
	d += "\tthis.sets.length: "+this.sets.length+"\n";
	d += "\tthis.terms.length: "+this.terms.length+"\n";
	dump(d);
}
