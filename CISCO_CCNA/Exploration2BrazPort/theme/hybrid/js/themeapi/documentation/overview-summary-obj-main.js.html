/**
* O objeto de nível mais alto que contém todos os dados e

* tema
* @constructor
*
* @author mpollack
* @version 1.0
*
*/
function Main() {
	dump("MAIN CONSTRUCTOR\n");
/**
* Nome fornecido para identificar o objeto na depuração 
* @type cadeia de caracteres
*/
	this.objname = "main";
/**
* Referência ao objeto de ferramentas globais 
* @type GlobalTools
*/
	this.globaltools = new GlobalTools(this);
/**
* Referência ao objeto de tema 
* @type Tema
*/
	this.theme = new Theme(this);
/**
* Referência ao objeto do glossário 
* @type Glossário
*/
	this.glossary = new Glossary(this);
/**
* Referência ao objeto do curso 
* @type Curso
*/
	this.course = new Course(this);
/**
* Referência ao objeto do capítulo atual 
* @type Capítulo
*/
	this.chapter = null;
/**
* Referência ao objeto da seção atual 
* @type Seção
*/
	this.section = null;
/**
* Referência ao objeto do tópico atual 
* @type Tópico
*/
	this.topic = null;
/**
* Referência ao objeto da página atual 
* @type Página
*/
	this.page = null;
	
}

Main.prototype.toggle = function() {
	this.globaltools = new GlobalTools(this);
	this.theme = new Theme(this);
	this.glossary = new Glossary(this);
	this.course = new Course(this);
	sendCommand("toggle");
}

/**
* envia o comando para obter as informações de página atuais
*/
Main.prototype.getCurrent = function() {
	sendCommand("current");
}

/**
* Define o capítulo de uma ID fornecida
* @param {string} i ID do capítulo
*/
Main.prototype.setCurrentChapterByID = function(i) {
	this.chapter = this.course.getChapterByID(i);
	this.chapter.dump();
}

/**
* Define a seção de uma ID fornecida
* @param {string} i ID da seção
*/
Main.prototype.setCurrentSectionByID = function(i) {
	this.section = this.chapter.getSectionByID(i);
	this.section.dump();
}

/**
* Define o tópico de uma ID fornecida
* @param {string} i ID de tópico
*/
Main.prototype.setCurrentTopicByID = function(i) {
	this.topic = this.section.getTopicByID(i);
	this.topic.dump();
}

/**
* Define a página de uma ID fornecida e se a página
* não tiver um texto definido, a chamada será feita
* para obter esse texto
* @param {string} i ID de página
*/
Main.prototype.setCurrentPageByID = function(i) {
	this.page = this.topic.getPageByID(i);
	if (this.page.contenttext == null) {
		this.page.getText();
	} else {
		this.page.dump();

}

/**
* Envia o comando de página anterior

*/
Main.prototype.previousPage = function() {
	sendCommand("prevpage");
}

/**
* Envia o comando de ir para a página informando o número LI fornecido
*

* @see #previousPage
* @see #nextPage
*/
Main.prototype.gotoPage = function(p) {
	sendCommand("gotopage", p);
