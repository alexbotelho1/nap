/**
* Contém dados de uma página
* @constructor
*
* @author mpollack
* @version 1.0
* @param {Topic} p referência ao objeto-pai
* @param {string} id ID de página
* @param {string} ty tipo de página
* @param {int} nu número da página
* @param {string} te texto da página
*
*/
function Page(p,id,ty,nu,te) {
/**
* Nome fornecido para identificar o objeto na depuração 
* @type cadeia de caracteres
*/
	this.objname = "page";
/**
* Referência ao objeto-pai
* @type Tópico
*/
	this.par = p;
/**
* ID da página
* @type cadeia de caracteres
*/
	this.oid = id;
/**
* Tipo de página
* @type cadeia de caracteres
*/
	this.type = ty;
/**
* Número da página
* @type int
*/
	this.num = nu;
/**
* Texto da página
* @type cadeia de caracteres
*/
	this.contenttext = new String(te);
/**
* Meio da página
* @type Meio
*/
	this.media = null;
/**
* Analisa o texto da página para converter links no formato apropriado ao uso em HTML
*/
	this.parseLinksInText();
}

/**
* Analisa o texto da página para converter links no formato apropriado ao uso em HTML
*/
Page.prototype.parseLinksInText = function() {
	if (this.contenttext.indexOf("asfunction:_global.core.gui.pglossary.getGlossaryTerm") > -1) {
		var s = this.contenttext.indexOf("asfunction:_global.core.gui.pglossary.getGlossaryTerm");
		var e = this.contenttext.indexOf("\"",s);
		var sub = this.contenttext.substring(s,e);
		var id = sub.substr(sub.indexOf(",")+1,sub.length);
		this.contenttext = this.contenttext.replace(sub,"javascript:top.popup('pglossary','"+id+"');");
		if (this.contenttext.indexOf("asfunction:_global.core.gui.pglossary.getGlossaryTerm") > -1) {
			this.parseLinksInText();
		}
	}
}

/**
* Envia o comando para obter o texto da página
*/
Page.prototype.getText = function() {
	sendCommand("pagetext");
}

/**
* Define o texto da página
* @param {string} d texto da página
*/
Page.prototype.setText = function(d) {
	this.contenttext = d;
	this.dump();
}

/**
* Envia informações de depuração de todas as propriedades para a janela de depuração
*/
Page.prototype.dump = function() {
	var d = "Page Dump\n";
	d += "\tthis.objname: "+this.objname+"\n";
	d += "\tthis.oid: "+this.oid+"\n";
	d += "\tthis.par: "+this.par+"\n";
	d += "\tthis.num: "+this.num+"\n";
	d += "\tthis.type: "+this.type+"\n";
	d += "\tthis.contenttext: "+this.contenttext+"\n";
	dump(d);
}
