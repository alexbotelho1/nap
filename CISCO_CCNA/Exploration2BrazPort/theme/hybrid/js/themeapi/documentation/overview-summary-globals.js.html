/**
* Contém variáveis e métodos de nível superior
*
* @author mpollack
* @version 1.0
*
*/

// mantém a referência ao objeto principal após a definição
main = null;
// mantém a referência ao objeto do evento após a definição
eventobj = null;
// booleano de determinação se todos os itens estiverem prontos
isready = false;
// booleano se deve ou não usar a janela de depuração
dumpon = false; 
// dados de depuração padrão
dumpdata = "";
// referência à janela de depuração
dumpwin = null;
// cria janela de depuração caso ela esteja ativada e não haja nenhuma
if (dumpon && !dumpwin) {
	dumpwin = window.open("../html/debug.html", "dumpwin");
}
/**
* Método para depurar informações se a depuração estiver ativada
* @param {string} d Informações a serem depuradas
*/
function dump(d) {
	if (dumpon) {
		dumpdata += datestamp()+"\t"+d;
		if (!dumpwin) {
			dumpwin = window.open("../html/debug.html", "dumpwin");
		} else {
			dumpwin.dump();
		}
	}
}
/**
* Método para limpar informações de depuração
*/
function cleardump() {
	dumpdata = "";
}

/**
* Método para obter carimbo de data/hora atual
* @returns {string} hora atual no formato hora:minuto:segundo:milissegundo
*/
function datestamp() {
	var d = new Date();
	var hou = d.getHours();
	var min = d.getMinutes();
	var sec = d.getSeconds();
	var mil = d.getMilliseconds();
	var r = hou+":"+min+":"+sec+":"+mil;
	return r;
}

/**
* Método definiu o conteúdo como pronto
*/
function setReady(s) {
	isready = s;
	dump("isready: "+isready+"\n");
	if (isready) {
		eventobj.eventprop = "contentready";
	}
}
