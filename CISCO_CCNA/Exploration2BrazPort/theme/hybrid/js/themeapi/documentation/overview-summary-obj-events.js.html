/**
* Mantém todas as referências de objeto do evento e os controladores
* @constructor
*
* @author mpollack
* @version 1.0
*
*/
function EventObj() {
/*
* conjunto this.watch depende do tipo de navegador.  Mozilla, naturalmente,
* oferece suporte à funcionalidade de observação.  IE exige sua adição.
* @type method
*/
	
	if (this.watch) {
		this.watch("eventprop", EventObj.prototype.eventWatcher);
	} else {
		this.ieWatch();
	}
/**
* variável de evento como observação
* @type boolean
*/
	this.eventprop = false;
/**
* Arranjo de eventos para ativação em comactive
* @type array
*/
	this.comactiveEventObj = new Array();
/**
* Arranjo de eventos para ativação em courseready
* @type array
*/
	this.coursereadyEventObj = new Array();
/**
* Arranjo de eventos para ativação em pageready
* @type array
*/
	this.pagereadyEventObj = new Array();
/**
* Arranjo de eventos para ativação em themeready
* @type array
*/
	this.themereadyEventObj = new Array();
/**
* Arranjo de eventos para ativação em toggle
* @type array
*/
	this.toggleEventObj = new Array();
/**
* Arranjo de eventos para ativação em globaltools
* @type array
*/
	this.globaltoolsreadyEventObj = new Array();
/**
* Arranjo de eventos para ativação em glossary
* @type array
*/
	this.glossaryreadyEventObj = new Array();
}
/**
* Adiciona evento a um arranjo de eventos com base no tipo.  Se não
* forem fornecidos dados o suficiente, um erro será lançado.
* @param {string} t Tipo de evento
* @param {method} e Evento a ser executado
* @param {string} i Identificador do evento
*/
EventObj.prototype.addEvent = function(t,e,i) {
	if (!t || !e || !i) {
		a = "ERRO: EventObj.addEvent – não há propriedades o suficiente\n";
		a+= "syntax:\tEventObj.addEvent(EventType, ItemToExecute, EventID);\n\n";
		a+= "\tEventObj.addEvent(\""+t+"\",\""+e+"\",\""+i+"\");\n";
		alert(a);
	} else {
		switch(t) {
			case 'courseready':
				this.coursereadyEventObj[i] = e;
				break;
			case 'comactive':
				this.comactiveEventObj[i] = e;
				break;
			case 'glossaryready':
				this.glossaryreadyEventObj[i] = e;
				break;
			case 'globaltoolsready':
				this.globaltoolsreadyEventObj[i] = e;
				break;
			case 'toggle':
				this.toggleEventObj[i] = e;
				break;
			case 'themeready':
				this.themereadyEventObj[i] = e;
				break;
			case 'pageready':
				this.pagereadyEventObj[i] = e;
				break;
			default:
				a = "ERRO: EventObj.addEvent - tratamento de evento incorreto\n";
				a+= "consulte a documentação para obter eventos válidos\n\n";
				a+= "\tEventObj.addEvent(\""+t+"\",\""+e+"\",\""+i+"\");\n";
				alert(a);
		}
	}
}
/**
* Remove um evento de um arranjo de eventos com base no tipo e na ID.  Se não
* forem fornecidos dados o suficiente, um erro será lançado.
* @param {string} t Tipo de evento
* @param {string} i Identificador do evento
*/
EventObj.prototype.removeEvent = function(t,i) {
	if (!t || !i) {
		a = "ERRO: EventObj.removeEvent - não há propriedades o suficiente\n";
		a+= "syntax:\tEventObj.removeEvent(EventType, EventID);\n\n";
		a+= "\tEventObj.removeEvent(\""+t+"\",\""+i+"\");\n";
		alert(a);
	} else {
		switch(t) {
			case 'courseready':
				this.coursereadyEventObj[i] = null;
				break;
			case 'comactive':
				this.comactiveEventObj[i] = null;
				break;
			case 'glossaryready':
				this.glossaryreadyEventObj[i] = null;
				break;
			case 'globaltoolsready':
				this.globaltoolsreadyEventObj[i] = null;
				break;
			case 'toggle':
				this.toggleEventObj[i] = null;
				break;
			case 'themeready':
				this.themereadyEventObj[i] = null;
				break;
			case 'pageready':
				this.pagereadyEventObj[i] = null;
				break;
			default:
				a = "ERRO: EventObj.removeEvent - tratamento de evento incorreto\n";
				a+= "consulte a documentação para obter eventos válidos\n\n";
				a+= "\tEventObj.addEvent(\""+t+"\",\""+e+"\",\""+i+"\");\n";
				alert(a);
		}
	}
}
/**
* Observa eventprop em busca de alterações.  Quando eventprop registra uma alteração
* um arranjo de evento é selecionado com base na propriedade newval.
* Em seguida, esse arranjo de eventos com loops é executado.  O
* eventprop é resetado em seguida.
* @param {string} prop Tipo de evento
* @param {string} oldval Identificador do evento
* @param {string} newval Identificador do evento
* @returns {boolean} Sempre retorna falso
*/
EventObj.prototype.eventWatcher = function(prop, oldval, newval) {
	eventarray = null;
	switch(newval) {
		case 'courseready':
			eventarray = this.coursereadyEventObj;
			break;
		case 'comactive':
			eventarray = this.comactiveEventObj;
			break;
		case 'pageready':
			eventarray = this.pagereadyEventObj;
			break;
		case 'globaltoolsready':
			eventarray = this.globaltoolsreadyEventObj;
			break;
		case 'toggle':
			eventarray = this.toggleEventObj;
			break;
		case 'themeready':
			eventarray = this.themereadyEventObj;
			break;
		case 'glossaryready':
			eventarray = this.glossaryreadyEventObj;
			break;
	}
	if (eventarray) {
		for (id in eventarray) {
			eval(eventarray[id]);
		}
	}
	return false;
}
/**
* Inicia a observação de IE
*/
EventObj.prototype.ieWatch = function() {
	setInterval(this.ieCheck,1000);
}
/**
* Processa observação para IE e, em caso de alteração,
* chama eventWatcher.
*/
EventObj.prototype.ieCheck = function() {
	var newval = eventobj.eventprop;
	//dump("ieCheck: "+eventobj+" "+newval+"\n");
	if (newval) {
		eventobj.eventWatcher("null", "null", newval);
	}
}


eventobj = new EventObj();
