/**
* Contém dados de um meio
* @constructor
*
* @author mpollack
* @version 1.0
* @param {Page} p referência ao objeto-pai
* @param {string} id ID de meio
* @param {string} ty tipo de meio
* @param {string} fi arquivo de meio
* @param {string} ex dados externos de meio
* @param {string} ft tipo de filtro de meio
*
*/
function Media(p,id,ty,fi,ex,ft) {
/**
* Nome fornecido para identificar o objeto na depuração 
* @type cadeia de caracteres
*/
	this.objname = "media";
/**
* Referência ao objeto-pai
* @type Página
*/
	this.par = p;
/**
* ID de meio
* @type cadeia de caracteres
*/
	this.oid = id;
/**
* Tipo de meio
* @type cadeia de caracteres
*/
	this.type = ty;
/**
* Referência do arquivo de meio
* @type cadeia de caracteres
*/
	this.file = fi;
/**
* Dados externos de meio
* @type cadeia de caracteres
*/
	this.external = ex;
/**
* Tipo de filtro de mídia
* @type cadeia de caracteres
*/
	this.filtertype = ft;
}

/**
* Envia informações de depuração de todas as propriedades para a janela de depuração
*/
Media.prototype.dump = function() {
	var d = "Media Dump\n";
	d += "\tthis.objname: "+this.objname+"\n";
	d += "\tthis.oid: "+this.oid+"\n";
	d += "\tthis.par: "+this.par+"\n";
	d += "\tthis.type: "+this.type+"\n";
	d += "\tthis.file: "+this.file+"\n";
	d += "\tthis.external: "+this.external+"\n";
	d += "\tthis.filtertype: "+this.filtertype+"\n";
	dump(d);
}
