/**
* Adiciona um item de tema ao arranjo associativo themetext
* @param {string} k chave de entrada
* @param {string} i item de entrada
*/
Theme.prototype.addThemeItem = function(k, i) {
	this.themetext[k] = i;
}
/**
* Retorna um item de tema do arranjo associativo themetext
* @param {string} i chave de entrada
* @returns {string} Cadeia de caracteres associada à chave fornecida ou uma cadeia de caracteres com erro
*/
Theme.prototype.getThemeItem = function(i) {
	if (this.themetext[i]) {
		r = this.themetext[i];
	} else {
		r = "ERROR: Nenhuma entrada para id "+i+" in theme data.";
	}
	return r;
}

/**
* Envia informações de depuração de todos os dados do tema para a janela de depuração
*/
Theme.prototype.dump = function() {
	var d = "Theme Dump\n";
	d += "\tthis.objname: "+this.objname+"\n";
	for (id in this.themetext) {
		d += "\tthis.themetext[\""+id+"\"]: "+this.getThemeItem(id)+"\n";
	}
	dump(d);
}
