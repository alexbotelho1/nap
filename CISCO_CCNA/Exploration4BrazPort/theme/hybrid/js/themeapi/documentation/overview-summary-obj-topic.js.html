/**
* Contém dados de um tópico
* @constructor
*
* @author mpollack
* @version 1.0
* @param {Section} p referência ao objeto-pai
* @param {string} id ID de tópico
* @param {string} ti título do tópico
* @param {string} ty tipo de tópico
* @param {int} nu número do tópico
*
*/
function Topic(p,id,ti,ty,nu) {
/**
* Nome fornecido para identificar o objeto na depuração 
* @type cadeia de caracteres
*/
	this.objname = "topic";
/**
* Referência ao objeto-pai
* @type Seção
*/
	this.par = p;
/**
* ID do tópico
* @type cadeia de caracteres
*/
	this.oid = id;
/**
* Título do tópico
* @type cadeia de caracteres
*/
	this.title = ti;
/**
* Tipo de tópico
* @type cadeia de caracteres
*/
	this.type = ty;
/**
* Número do tópico
* @type int
*/
	this.num = nu;
/**
* Objetos de página em um arranjo seqüencial
* @type arranjo
*/
	this.pages = new Array();
/**
* Objetos de tópico em um arranjo associativo
* @type arranjo
*/
	this.pagesID = new Array();
}

/**
* Obtém a página de uma ID fornecida
* @param {string} i ID de página
* @return A {@link Page} instância
* @type Página
*/
Topic.prototype.getPageByID = function(i) {
	var r = this.pagesID[i];
	return r;
}

/**
* Verifica o tópico para ver se há mais alguma página na árvore
* contêm o meio de um determinado tipo
* @param {string} f tipo de filtro de meio
* @return booleano
* @type Booleano
*/
Topic.prototype.containsMediaByFilter = function(f) {
	var check = false;
	for (var j=0; j<this.pages.length; j++) {
		if (this.pages[j].media.filtertype == f) {
			check = true;
			j = this.pages.length;
		}
	}
	return check;
}


/**
* Envia informações de depuração de todas as propriedades para a janela de depuração
*/
Topic.prototype.dump = function() {
	var d = "Topic Dump\n";
	d += "\tthis.objname: "+this.objname+"\n";
	d += "\tthis.oid: "+this.oid+"\n";
	d += "\tthis.par: "+this.par+"\n";
	d += "\tthis.title: "+this.title+"\n";
	d += "\tthis.num: "+this.num+"\n";
	d += "\tthis.type: "+this.type+"\n";
	d += "\tthis.pages: "+this.pages+"\n";
	dump(d);
}
